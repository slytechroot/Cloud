In the previous video we wrote a Jenkins file and understand the syntax.

Now we can write a Jenkins file for our pipeline which does the same as our Freestyle job. You can check it out to see the steps. 
So let's write this as a Jenkins file.

########

pipeline {
	agent any
	tools {
		maven "Maven"
	}
	stages {
		stage("build jar"){
			steps {
				script {
					echo "building the application..."
					sh "mvn package"
				}
			}
		}
		stage("build image"){
			steps {
				script {
					echo "building the application..."
					withCredentials([usernamePassword(credentialsID: 'docker-hub-repo', passwordVarialble: 'PASS', usernameVariable: 'USER')]){
						sh 'docker build -t sly/demo-app:jma-2.0 .'
						sh "echo $PASS | docker login -u $USER --password-stdin"
						sh 'docker push sly/demo-app:jma-2.0'
					}						
				}
			}
		}
		stage("deploy"){
			steps {
				script {
					echo "deploying the application..."
				}
			}
		}
		
###############
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	
	}
}
